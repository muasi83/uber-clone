class User {
  final int? id;
  final String username;
  final String email;
  final String password;
  final String fullName;
  final String? deviceToken;
  bool isOnline;  // ✅ CHANGED FROM final TO mutable
  final DateTime createdAt;

  User({
    this.id,
    required this.username,
    required this.email,
    required this.password,
    required this.fullName,
    this.deviceToken,
    this.isOnline = false,
    DateTime? createdAt,
  }) : createdAt = createdAt ?? DateTime.now();

  factory User.fromJson(Map<String, dynamic> json) => User(
        id: json['id'] as int?,
        username: json['username'] as String,
        email: json['email'] as String,
        password: json['password'] as String,
        fullName: json['fullName'] as String,
        deviceToken: json['deviceToken'] as String?,
        isOnline: json['isOnline'] is bool 
            ? json['isOnline'] as bool 
            : (json['isOnline'] as int?) == 1,
        createdAt: json['createdAt'] != null
            ? DateTime.parse(json['createdAt'] as String)
            : DateTime.now(),
      );

  Map<String, dynamic> toJson() => {
        'id': id,
        'username': username,
        'email': email,
        'password': password,
        'fullName': fullName,
        'deviceToken': deviceToken,
        'isOnline': isOnline ? 1 : 0,
        'createdAt': createdAt.toIso8601String(),
      };

  User copyWith({
    int? id,
    String? username,
    String? email,
    String? password,
    String? fullName,
    String? deviceToken,
    bool? isOnline,
    DateTime? createdAt,
  }) {
    return User(
      id: id ?? this.id,
      username: username ?? this.username,
      email: email ?? this.email,
      password: password ?? this.password,
      fullName: fullName ?? this.fullName,
      deviceToken: deviceToken ?? this.deviceToken,
      isOnline: isOnline ?? this.isOnline,
      createdAt: createdAt ?? this.createdAt,
    );
  }
}
class Message {
  final int? id;
  final int senderId;
  final int receiverId;
  final String content;
  final DateTime sentAt;
  final bool isRead;

  Message({
    this.id,
    required this.senderId,
    required this.receiverId,
    required this.content,
    DateTime? sentAt,
    this.isRead = false,
  }) : sentAt = sentAt ?? DateTime.now();

  factory Message.fromJson(Map<String, dynamic> json) => Message(
        id: json['id'] as int?,
        senderId: json['senderId'] as int,
        receiverId: json['receiverId'] as int,
        content: json['content'] as String,
        sentAt: json['sentAt'] != null
            ? DateTime.parse(json['sentAt'] as String)
            : DateTime.now(),
        isRead: (json['isRead'] as int?) == 1,
      );

  Map<String, dynamic> toJson() => {
        'id': id,
        'senderId': senderId,
        'receiverId': receiverId,
        'content': content,
        'sentAt': sentAt.toIso8601String(),
        'isRead': isRead ? 1 : 0,
      };

  Message copyWith({
    int? id,
    int? senderId,
    int? receiverId,
    String? content,
    DateTime? sentAt,
    bool? isRead,
  }) {
    return Message(
      id: id ?? this.id,
      senderId: senderId ?? this.senderId,
      receiverId: receiverId ?? this.receiverId,
      content: content ?? this.content,
      sentAt: sentAt ?? this.sentAt,
      isRead: isRead ?? this.isRead,
    );
  }
}

import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;
import 'dart:convert';

class AuthScreen extends StatefulWidget {
  const AuthScreen({Key? key}) : super(key: key);

  @override
  State<AuthScreen> createState() => _AuthScreenState();
}

class _AuthScreenState extends State<AuthScreen> {
  final _usernameController = TextEditingController();
  final _emailController = TextEditingController();
  final _passwordController = TextEditingController();
  final _fullNameController = TextEditingController();
  bool _isLogin = true;
  bool _isLoading = false;

  Future<void> _handleAuth() async {
    if (_isLogin) {
      await _login();
    } else {
      await _register();
    }
  }

  Future<void> _register() async {
    setState(() => _isLoading = true);
    try {
      final url = StorageService.getAuthRegisterUrl();
      
      // DEBUG: Print the URL
      addDebugMessage('═══════════════════════════════════════');
      addDebugMessage('🔍 DEBUG REGISTER:');
      addDebugMessage('URL: $url');
      addDebugMessage('Username: ${_usernameController.text}');
      addDebugMessage('Email: ${_emailController.text}');
      addDebugMessage('Full Name: ${_fullNameController.text}');
      addDebugMessage('═══════════════════════════════════════');

      addDebugMessage('🚀 Sending HTTP POST request...');
      
      final response = await http
          .post(
            Uri.parse(url),
            headers: {'Content-Type': 'application/json'},
            body: jsonEncode({
              'username': _usernameController.text,
              'email': _emailController.text,
              'password': _passwordController.text,
              'fullName': _fullNameController.text,
            }),
          )
          .timeout(
            const Duration(seconds: 30),
            onTimeout: () {
              addDebugMessage('❌ TIMEOUT: Request took too long');
              throw Exception('Server not responding');
            },
          );

      addDebugMessage('✅ Response received');
      addDebugMessage('Status Code: ${response.statusCode}');
      addDebugMessage('Response Body: ${response.body}');

      if (response.statusCode == 200) {
        final json = jsonDecode(response.body);
        addDebugMessage('✅ Registration successful: ${json['message']}');
        
        if (mounted) {
          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(
              content: Text(json['message']),
              backgroundColor: Colors.green,
            ),
          );
          setState(() => _isLogin = true);
          _clearFields();
        }
      } else {
        addDebugMessage('❌ Register failed with status: ${response.statusCode}');
        if (mounted) {
          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(
              content: Text('Register failed: ${response.body}'),
              backgroundColor: Colors.red,
            ),
          );
        }
      }
    } catch (e) {
      addDebugMessage('❌ EXCEPTION IN REGISTER:');
      addDebugMessage('Error type: ${e.runtimeType}');
      addDebugMessage('Error message: $e');
      addDebugMessage('═══════════════════════════════════════');
      
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(
            content: Text('Error: $e'),
            backgroundColor: Colors.red,
          ),
        );
      }
    } finally {
      setState(() => _isLoading = false);
    }
  }

  Future<void> _login() async {
    setState(() => _isLoading = true);
    try {
      final url = StorageService.getAuthLoginUrl();
      
      // DEBUG: Print the URL
      addDebugMessage('═══════════════════════════════════════');
      addDebugMessage('🔍 DEBUG LOGIN:');
      addDebugMessage('URL: $url');
      addDebugMessage('Username: ${_usernameController.text}');
      addDebugMessage('═══════════════════════════════════════');

      addDebugMessage('🚀 Sending HTTP POST request...');
      
      final response = await http
          .post(
            Uri.parse(url),
            headers: {'Content-Type': 'application/json'},
            body: jsonEncode({
              'username': _usernameController.text,
              'password': _passwordController.text,
            }),
          )
          .timeout(
            const Duration(seconds: 30),
            onTimeout: () {
              addDebugMessage('❌ TIMEOUT: Request took too long');
              throw Exception('Server not responding');
            },
          );

      addDebugMessage('✅ Response received');
      addDebugMessage('Status Code: ${response.statusCode}');
      addDebugMessage('Response Body: ${response.body}');

    if (response.statusCode == 200) {
  final json = jsonDecode(response.body);
  addDebugMessage('✅ Login successful');
  
  // ✅ Connect WebSocket and send login
  await Future.delayed(Duration(milliseconds: 500));
  WebSocketService.setOnline(json['userId'], json['username']);
  
  if (mounted) {
    Navigator.of(context).pushReplacement(
            MaterialPageRoute(
              builder: (context) => HomeScreen(
                userId: json['userId'],
                username: json['username'],
                token: json['token'],
              ),
            ),
          );
        }
      } else {
        addDebugMessage('❌ Login failed with status: ${response.statusCode}');
        if (mounted) {
          ScaffoldMessenger.of(context).showSnackBar(
            const SnackBar(
              content: Text('Login failed'),
              backgroundColor: Colors.red,
            ),
          );
        }
      }
    } catch (e) {
      addDebugMessage('❌ EXCEPTION IN LOGIN:');
      addDebugMessage('Error type: ${e.runtimeType}');
      addDebugMessage('Error message: $e');
      addDebugMessage('═══════════════════════════════════════');
      
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(
            content: Text('Error: $e'),
            backgroundColor: Colors.red,
          ),
        );
      }
    } finally {
      setState(() => _isLoading = false);
    }
  }

  void _clearFields() {
    _usernameController.clear();
    _emailController.clear();
    _passwordController.clear();
    _fullNameController.clear();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(_isLogin ? 'Login' : 'Register'),
        centerTitle: true,
        actions: [
  IconButton(
    icon: const Icon(Icons.bug_report),
    onPressed: () {
      Navigator.of(context).push(
        MaterialPageRoute(
          builder: (context) => const DebugScreen(),
        ),
      );
    },
  ),
  IconButton(
    icon: const Icon(Icons.settings),
    onPressed: () {
      Navigator.of(context).push(
        MaterialPageRoute(
          builder: (context) => const SettingsScreen(
            username: 'Guest',
          ),
        ),
      );
    },
  ),
],
      ),
      body: SingleChildScrollView(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [
            const SizedBox(height: 40),
            Icon(
              Icons.chat_bubble_outline,
              size: 80,
              color: Colors.blue[400],
            ),
            const SizedBox(height: 30),
            const Text(
              'Decentralized Chat',
              style: TextStyle(
                fontSize: 28,
                fontWeight: FontWeight.bold,
              ),
            ),
            const SizedBox(height: 40),
            if (!_isLogin)
              TextField(
                controller: _fullNameController,
                decoration: InputDecoration(
                  labelText: 'Full Name',
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(8),
                  ),
                  prefixIcon: const Icon(Icons.person),
                ),
              ),
            if (!_isLogin) const SizedBox(height: 16),
            TextField(
              controller: _usernameController,
              decoration: InputDecoration(
                labelText: 'Username',
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(8),
                ),
                prefixIcon: const Icon(Icons.account_circle),
              ),
            ),
            const SizedBox(height: 16),
            if (!_isLogin)
              TextField(
                controller: _emailController,
                decoration: InputDecoration(
                  labelText: 'Email',
                  border: OutlineInputBorder(
                    borderRadius: BorderRadius.circular(8),
                  ),
                  prefixIcon: const Icon(Icons.email),
                ),
              ),
            if (!_isLogin) const SizedBox(height: 16),
            TextField(
              controller: _passwordController,
              obscureText: true,
              decoration: InputDecoration(
                labelText: 'Password',
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(8),
                ),
                prefixIcon: const Icon(Icons.lock),
              ),
            ),
            const SizedBox(height: 24),
            SizedBox(
              width: double.infinity,
              child: ElevatedButton(
                onPressed: _isLoading ? null : _handleAuth,
                style: ElevatedButton.styleFrom(
                  padding: const EdgeInsets.symmetric(vertical: 14),
                ),
                child: _isLoading
                    ? const SizedBox(
                        height: 20,
                        width: 20,
                        child: CircularProgressIndicator(
                          strokeWidth: 2,
                          valueColor:
                              AlwaysStoppedAnimation<Color>(Colors.white),
                        ),
                      )
                    : Text(_isLogin ? 'Login' : 'Register'),
              ),
            ),
            const SizedBox(height: 16),
            TextButton(
              onPressed: () {
                setState(() => _isLogin = !_isLogin);
                _clearFields();
              },
              child: Text(_isLogin
                  ? 'Don\'t have account? Register'
                  : 'Already have account? Login'),
            ),
          ],
        ),
      ),
    );
  }

  @override
  void dispose() {
    _usernameController.dispose();
    _emailController.dispose();
    _passwordController.dispose();
    _fullNameController.dispose();
    super.dispose();
  }
}



import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;
import 'dart:async';
import 'dart:convert';
import '../models/models.dart';

import '../services/websocket_service.dart';
import 'debug_screen.dart';

class ChatScreen extends StatefulWidget {
  final int currentUserId;
  final String currentUsername;
  final int receiverId;
  final String receiverName;
  final String token;

  const ChatScreen({
    Key? key,
    required this.currentUserId,
    required this.currentUsername,
    required this.receiverId,
    required this.receiverName,
    required this.token,
  }) : super(key: key);

  @override
  State<ChatScreen> createState() => _ChatScreenState();
}

class _ChatScreenState extends State<ChatScreen> {
  final _messageController = TextEditingController();
  List<Message> messages = [];
  bool isLoading = true;
  late Timer _refreshTimer;

  @override
  void initState() {
    super.initState();
    _loadChatHistory();
    
    // ✅ Refresh messages every 2 seconds (fallback)
    _refreshTimer = Timer.periodic(Duration(seconds: 2), (_) {
      _loadChatHistory();
    });
    
    // ✅ Listen for WebSocket messages
    WebSocketService.onMessageReceived = _handleIncomingMessage;
  }

  @override
  void dispose() {
    _messageController.dispose();
    _refreshTimer.cancel();
    super.dispose();
  }

  // ✅ Handle incoming message via WebSocket
  void _handleIncomingMessage(Map<String, dynamic> message) {
    final senderId = message['senderId'];
    final receiverId = message['receiverId'];
    
    // Only add if it's for this conversation
    if ((senderId == widget.receiverId && receiverId == widget.currentUserId) ||
        (senderId == widget.currentUserId && receiverId == widget.receiverId)) {
      
      print('✅ New message received via WebSocket!');
      
      final newMessage = Message(
        senderId: senderId,
        receiverId: receiverId,
        content: message['content'] ?? '',
        sentAt: DateTime.now(),
        isRead: false,
      );
      
      setState(() {
        messages.add(newMessage);
      });
      
      // Show notification
      _showMessageNotification(newMessage);
    }
  }

  Future<void> _loadChatHistory() async {
    try {
      final baseUrl = StorageService.getChatHistoryUrl();
      final url =
          '$baseUrl?userId1=${widget.currentUserId}&userId2=${widget.receiverId}';

      addDebugMessage('🔍 Loading chat history from: $url');

      final response = await http.get(
        Uri.parse(url),
      ).timeout(
        const Duration(seconds: 10),
        onTimeout: () => throw Exception('Server timeout'),
      );

      addDebugMessage('✅ Response status: ${response.statusCode}');

      if (response.statusCode == 200) {
        final jsonList = jsonDecode(response.body) as List;
        
        addDebugMessage('📊 Parsed ${jsonList.length} messages');

        setState(() {
          messages = jsonList
              .map((m) {
                try {
                  return Message.fromJson(m);
                } catch (e) {
                  addDebugMessage('❌ Error parsing message: $e');
                  return null;
                }
              })
              .whereType<Message>()
              .toList();
          isLoading = false;
        });

        addDebugMessage('✅ Chat history loaded: ${messages.length} messages');
      } else {
        addDebugMessage('❌ Server returned: ${response.statusCode}');
        setState(() => isLoading = false);
      }
    } catch (e) {
      addDebugMessage('❌ Error loading chat: $e');
      setState(() => isLoading = false);
    }
  }

  Future<void> _sendMessage() async {
    if (_messageController.text.isEmpty) return;

    final content = _messageController.text;
    _messageController.clear();

    try {
      final url = StorageService.getChatSendUrl();

      addDebugMessage('🚀 Sending message to: $url');
      addDebugMessage('Content: $content');

      final response = await http
          .post(
            Uri.parse(url),
            headers: {
              'Content-Type': 'application/json',
              'Authorization': 'Bearer ${widget.token}',
            },
            body: jsonEncode({
              'senderId': widget.currentUserId,
              'receiverId': widget.receiverId,
              'content': content,
            }),
          )
          .timeout(
            const Duration(seconds: 10),
            onTimeout: () => throw Exception('Server timeout'),
          );

      addDebugMessage('✅ Message sent, status: ${response.statusCode}');

      // ✅ Also send via WebSocket for real-time
      WebSocketService.sendMessage(
        widget.currentUserId,
        widget.receiverId,
        content,
        widget.currentUsername,
      );

      if (response.statusCode == 200) {
        // Refresh chat after sending
        await Future.delayed(Duration(milliseconds: 500));
        _loadChatHistory();
      } else {
        addDebugMessage('❌ Failed to send: ${response.body}');
        if (mounted) {
          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(content: Text('Failed to send message')),
          );
        }
      }
    } catch (e) {
      addDebugMessage('❌ Error: $e');
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(content: Text('Error: $e')),
        );
      }
    }
  }

  void _showMessageNotification(Message message) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Text('💬 ${widget.receiverName}: ${message.content}'),
        duration: const Duration(seconds: 3),
        backgroundColor: Colors.blue,
      ),
    );
  }
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(widget.receiverName),
        centerTitle: true,
        actions: [
          IconButton(
            icon: const Icon(Icons.bug_report),
            onPressed: () {
              Navigator.of(context).push(
                MaterialPageRoute(
                  builder: (context) => const DebugScreen(),
                ),
              );
            },
          ),
        ],
      ),
      body: Column(
        children: [
          Expanded(
            child: isLoading
                ? const Center(child: CircularProgressIndicator())
                : messages.isEmpty
                    ? const Center(
                        child: Text('No messages yet. Start chatting!'),
                      )
                    : ListView.builder(
                        reverse: false,
                        itemCount: messages.length,
                        itemBuilder: (context, index) {
                          final message = messages[index];
                          final isSent =
                              message.senderId == widget.currentUserId;
                          return Align(
                            alignment: isSent
                                ? Alignment.centerRight
                                : Alignment.centerLeft,
                            child: Container(
                              margin: const EdgeInsets.all(8),
                              padding: const EdgeInsets.all(12),
                              decoration: BoxDecoration(
                                color: isSent ? Colors.blue : Colors.grey[300],
                                borderRadius: BorderRadius.circular(12),
                              ),
                              child: Column(
                                crossAxisAlignment: CrossAxisAlignment.start,
                                children: [
                                  Text(
                                    message.content,
                                    style: TextStyle(
                                      color: isSent ? Colors.white : Colors.black,
                                      fontSize: 16,
                                    ),
                                  ),
                                  const SizedBox(height: 4),
                                  Text(
                                    message.sentAt.toString().split('.').first,
                                    style: TextStyle(
                                      color: isSent
                                          ? Colors.white70
                                          : Colors.black54,
                                      fontSize: 12,
                                    ),
                                  ),
                                ],
                              ),
                            ),
                          );
                        },
                      ),
          ),
          Padding(
            padding: const EdgeInsets.all(8),
            child: Row(
              children: [
                Expanded(
                  child: TextField(
                    controller: _messageController,
                    decoration: InputDecoration(
                      hintText: 'Type a message...',
                      border: OutlineInputBorder(
                        borderRadius: BorderRadius.circular(20),
                      ),
                      contentPadding: const EdgeInsets.symmetric(
                        horizontal: 16,
                        vertical: 12,
                      ),
                    ),
                    maxLines: null,
                  ),
                ),
                const SizedBox(width: 8),
                FloatingActionButton(
                  onPressed: _sendMessage,
                  mini: true,
                  child: const Icon(Icons.send),
                ),
              ],
            ),
          ),
        ],
      ),
    );
  }
}


import 'package:flutter/material.dart';

// Global list to store debug messages
List<String> debugMessages = [];

// Function to add debug message
void addDebugMessage(String message) {
  debugMessages.add('[${DateTime.now().hour}:${DateTime.now().minute}:${DateTime.now().second}] $message');
  print(message); // Also print to console
}

class DebugScreen extends StatefulWidget {
  const DebugScreen({Key? key}) : super(key: key);

  @override
  State<DebugScreen> createState() => _DebugScreenState();
}

class _DebugScreenState extends State<DebugScreen> {
  late ScrollController _scrollController;

  @override
  void initState() {
    super.initState();
    _scrollController = ScrollController();
    _scrollToBottom();
  }

  void _scrollToBottom() {
    WidgetsBinding.instance.addPostFrameCallback((_) {
      if (_scrollController.hasClients) {
        _scrollController.jumpTo(_scrollController.position.maxScrollExtent);
      }
    });
  }

  @override
  Widget build(BuildContext context) {
    _scrollToBottom();
    
    return Scaffold(
      appBar: AppBar(
        title: const Text('Debug Console'),
        centerTitle: true,
        actions: [
          IconButton(
            icon: const Icon(Icons.delete),
            tooltip: 'Clear',
            onPressed: () {
              setState(() {
                debugMessages.clear();
              });
            },
          ),
          IconButton(
            icon: const Icon(Icons.refresh),
            tooltip: 'Refresh',

  try {

    final response = await http.get(Uri.parse(url)).timeout(
      const Duration(seconds: 5),
      onTimeout: () => throw Exception('Timeout'),
    );

    if (response.statusCode == 200) {
      final messages = jsonDecode(response.body) as List;
      
      // Count messages from other user that are NOT read
      int unreadCount = 0;
      for (var msg in messages) {
        if (msg['senderId'] == otherUserId && msg['isRead'] == 0) {
          unreadCount++;
        }
      }
      
      addDebugMessage('📊 Unread from user $otherUserId: $unreadCount');
      return unreadCount;
    }
    return 0;
  } catch (e) {

  Future<shelf.Response> _handleLogin(shelf.Request request) async {
    print('═══════════════════════════════════════');
    print('🔍 LOGIN REQUEST RECEIVED');
    print('═══════════════════════════════════════');
    
    try {
      print('📖 Reading request body...');
      final body = await request.readAsString();
      print('📖 Body: $body');
      
      final json = jsonDecode(body);
      print('✅ JSON decoded');

      print('🔎 Finding user by username: ${json['username']}');
      final user = await dbService.findUserByUsername(json['username']);

      if (user == null) {
        print('❌ User not found');
        return shelf.Response(400,
            body: jsonEncode({'message': 'User not found'}),
            headers: {'Content-Type': 'application/json'});
      }
      
      print('✅ User found: ${user.username}');

      print('🔐 Checking password...');
      if (user.password != _hashPassword(json['password'])) {
        print('❌ Password incorrect');
        return shelf.Response(400,
            body: jsonEncode({'message': 'Invalid password'}),
            headers: {'Content-Type': 'application/json'});
      }
      
      print('✅ Password correct');

      print('🔐 Generating token...');
      final token = _generateToken(user.id!, user.username);
      print('✅ Token generated');

      print('✅ LOGIN SUCCESSFUL');
      print('═══════════════════════════════════════');
      
      return shelf.Response.ok(
        jsonEncode({
          'token': token,
          'userId': user.id,
          'username': user.username,
          'message': 'Login successful',
        }),
        headers: {'Content-Type': 'application/json'},
      );
    } catch (e) {
      print('❌ ERROR IN LOGIN: $e');
      print('═══════════════════════════════════════');
      
      return shelf.Response(400,
          body: jsonEncode({'message': e.toString()}),
          headers: {'Content-Type': 'application/json'});
    }
  }

  Future<shelf.Response> _handleDeviceToken(shelf.Request request) async {
    print('═══════════════════════════════════════');
    print('🔍 DEVICE TOKEN REQUEST RECEIVED');
    print('═══════════════════════════════════════');
    
    try {
      final token = request.headers['authorization']?.replaceFirst('Bearer ', '');
      if (token == null) {
        print('❌ No authorization header');
        return shelf.Response(401,
            body: jsonEncode({'message': 'Unauthorized'}),
            headers: {'Content-Type': 'application/json'});
      }

      final userId = _getUserIdFromToken(token);
      if (userId == null) {
        print('❌ Invalid token');
        return shelf.Response(401,
            body: jsonEncode({'message': 'Invalid token'}),
            headers: {'Content-Type': 'application/json'});
      }

      final body = await request.readAsString();
      final json = jsonDecode(body);

      await dbService.updateDeviceToken(userId, json['deviceToken']);
      print('✅ Device token updated for user: $userId');

      return shelf.Response.ok(
        jsonEncode({'message': 'Device token updated'}),
        headers: {'Content-Type': 'application/json'},
      );
    } catch (e) {
      print('❌ ERROR IN DEVICE TOKEN: $e');
      return shelf.Response(400,
          body: jsonEncode({'message': e.toString()}),

  // CHAT HANDLERS
  Future<shelf.Response> _handleSendMessage(shelf.Request request) async {
    print('🔍 SEND MESSAGE REQUEST');
    try {
      final body = await request.readAsString();
      final json = jsonDecode(body);

      final message = Message(
        senderId: json['senderId'],
        receiverId: json['receiverId'],
        content: json['content'],
      );

      await dbService.saveMessage(message);
      print('✅ Message saved from ${message.senderId} to ${message.receiverId}');

      // Send notification
      final receiver = await dbService.findUserById(message.receiverId);
      if (receiver != null) {
        final sender = await dbService.findUserById(message.senderId);
        final notification = AppNotification(
          userId: message.receiverId,
          title: sender?.fullName ?? 'Unknown',
          body: message.content,
          type: 'message',
        );
        await dbService.saveNotification(notification);
        print('✅ Notification saved');
      }

      return shelf.Response.ok(
        jsonEncode({'message': 'Message sent'}),
        headers: {'Content-Type': 'application/json'},
      );
    } catch (e) {
      print('❌ ERROR IN SEND MESSAGE: $e');
      return shelf.Response(400,

  Future<shelf.Response> _handleGetChatHistory(shelf.Request request) async {

      final userId2 = int.parse(request.url.queryParameters['userId2'] ?? '0');

  // NOTIFICATION HANDLERS
  Future<shelf.Response> _handleGetNotifications(shelf.Request request) async {
    print('🔍 GET NOTIFICATIONS REQUEST');
    try {
      final token = request.headers['authorization']?.replaceFirst('Bearer ', '');
      if (token == null) {
        print('❌ No authorization');
        return shelf.Response(401,
            body: jsonEncode({'message': 'Unauthorized'}),
            headers: {'Content-Type': 'application/json'});
      }

      final userId = _getUserIdFromToken(token);
      if (userId == null) {
        print('❌ Invalid token');
        return shelf.Response(401,
            body: jsonEncode({'message': 'Invalid token'}),
            headers: {'Content-Type': 'application/json'});
      }

      final notifications = await dbService.getUserNotifications(userId);
      print('✅ Found ${notifications.length} notifications for user $userId');

      return shelf.Response.ok(
        jsonEncode(notifications.map((n) => n.toJson()).toList()),
        headers: {'Content-Type': 'application/json'},
      );
    } catch (e) {
      print('❌ ERROR IN GET NOTIFICATIONS: $e');
      return shelf.Response(400,
          body: jsonEncode({'message': e.toString()}),
          headers: {'Content-Type': 'application/json'});
    }
  }

  Future<shelf.Response> _handleMarkNotificationAsRead(
      shelf.Request request, String notificationId) async {
    print('🔍 MARK NOTIFICATION READ REQUEST: $notificationId');
    try {
      await dbService.markNotificationAsRead(int.parse(notificationId));
      print('✅ Notification marked as read');
      return shelf.Response.ok(

        headers: {'Content-Type': 'application/json'},

    } catch (e) {

      return shelf.Response(400,
          body: jsonEncode({'message': e.toString()}),
          headers: {'Content-Type': 'application/json'});
    }
  }

  // UTILITY FUNCTIONS
  String _hashPassword(String password) {
    return sha256.convert(password.codeUnits).toString();
  }

  String _generateToken(int userId, String username) {
    final payload = {
      'userId': userId,
      'username': username,
      'iat': DateTime.now().millisecondsSinceEpoch ~/ 1000,
      'exp': (DateTime.now()
                  .add(const Duration(hours: 24))
                  .millisecondsSinceEpoch ~/
              1000),
    };
    return base64Url.encode(utf8.encode(jsonEncode(payload)));
  }

  int? _getUserIdFromToken(String token) {
    try {
      final decoded =
          jsonDecode(utf8.decode(base64Url.decode(token.split('.').last)));
      return decoded['userId'];
    } catch (e) {
      return null;
    }
  }

  Future<void> start() async {
    print('═══════════════════════════════════════');
    print('🚀 Starting backend server on port $port...');
    print('⏰ Time: ${DateTime.now()}');
    print('═══════════════════════════════════════');
    
    try {
      _server = await shelf_io.serve(
        _router,
        '0.0.0.0',
        port,
      );
      print('✅ HTTP Server created successfully');
      print('✅ Server listening on: 0.0.0.0:$port');
      print('✅ Server running and ready to accept connections');
      print('═══════════════════════════════════════');
    } catch (e) {
      print('❌ SERVER START ERROR');
      print('❌ Error: $e');
      print('❌ Type: ${e.runtimeType}');
      print('═══════════════════════════════════════');
    }
  }

  Future<void> stop() async {
    print('🛑 Stopping server...');
    if (_server != null) {
      await _server!.close();
      print('✅ Server stopped');
    }
  }
}

import 'package:sqflite/sqflite.dart';
import 'package:path/path.dart';
import '../models/models.dart';

class DatabaseService {
  static final DatabaseService _instance = DatabaseService._internal();

  factory DatabaseService() {
    return _instance;
  }

  DatabaseService._internal();

  static Database? _database;

  Future<Database> get database async {
    if (_database != null) return _database!;
    _database = await _initDatabase();
    return _database!;
  }

  Future<Database> _initDatabase() async {
    String path = join(await getDatabasesPath(), 'chat_app.db');
    return await openDatabase(
      path,
      version: 1,
      onCreate: _onCreate,
    );
  }

  Future<void> _onCreate(Database db, int version) async {
    await db.execute('''
      CREATE TABLE users(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        username TEXT UNIQUE NOT NULL,
        email TEXT UNIQUE NOT NULL,
        password TEXT NOT NULL,
        fullName TEXT NOT NULL,
        deviceToken TEXT,
        isOnline INTEGER NOT NULL,
        createdAt TEXT NOT NULL
      )
    ''');

    await db.execute('''
      CREATE TABLE messages(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        senderId INTEGER NOT NULL,
        receiverId INTEGER NOT NULL,
        content TEXT NOT NULL,
        sentAt TEXT NOT NULL,
        isRead INTEGER NOT NULL,
        FOREIGN KEY(senderId) REFERENCES users(id),
        FOREIGN KEY(receiverId) REFERENCES users(id)
      )
    ''');

    await db.execute('''
      CREATE TABLE notifications(
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        userId INTEGER NOT NULL,
        title TEXT NOT NULL,
        body TEXT NOT NULL,
        type TEXT NOT NULL,
        isRead INTEGER NOT NULL,
        createdAt TEXT NOT NULL,
        relatedUserId TEXT,
        FOREIGN KEY(userId) REFERENCES users(id)
      )
    ''');
  }

  // USER OPERATIONS
  Future<User?> registerUser(User user) async {
    try {
      final db = await database;
      final id = await db.insert('users', {
        'username': user.username,
        'email': user.email,
        'password': user.password,
        'fullName': user.fullName,
        'isOnline': user.isOnline ? 1 : 0,
        'createdAt': user.createdAt.toIso8601String(),
      });
      return user.copyWith(id: id);
    } catch (e) {
      print('Error registering user: $e');
      return null;
    }
  }

  Future<User?> findUserByUsername(String username) async {
    final db = await database;
    final result = await db.query(
      'users',
      where: 'username = ?',
      whereArgs: [username],
    );
    if (result.isEmpty) return null;
    return User.fromJson(result.first);
  }

  Future<User?> findUserById(int id) async {
    final db = await database;
    final result = await db.query(
      'users',
      where: 'id = ?',
      whereArgs: [id],
    );
    if (result.isEmpty) return null;
    return User.fromJson(result.first);
  }

  Future<List<User>> getAllUsers() async {
    final db = await database;
    final result = await db.query('users');
    return result.map((json) => User.fromJson(json)).toList();
  }

  Future<void> updateDeviceToken(int userId, String token) async {
    final db = await database;
    await db.update(
      'users',
      {'deviceToken': token},
      where: 'id = ?',
      whereArgs: [userId],
    );
  }

  Future<void> setUserOnline(int userId, bool online) async {
    final db = await database;
    await db.update(
      'users',
      {'isOnline': online ? 1 : 0},
      where: 'id = ?',
      whereArgs: [userId],
    );
  }

  // MESSAGE OPERATIONS
  Future<Message> saveMessage(Message message) async {
    final db = await database;
    final id = await db.insert('messages', {
      'senderId': message.senderId,
      'receiverId': message.receiverId,
      'content': message.content,
      'sentAt': message.sentAt.toIso8601String(),
      'isRead': message.isRead ? 1 : 0,
    });
    return message.copyWith(id: id);
  }

  Future<List<Message>> getChatHistory(int userId1, int userId2) async {
    final db = await database;
    final result = await db.query(
      'messages',
      where:
          '(senderId = ? AND receiverId = ?) OR (senderId = ? AND receiverId = ?)',
      whereArgs: [userId1, userId2, userId2, userId1],
      orderBy: 'sentAt DESC',
    );
    return result.map((json) => Message.fromJson(json)).toList();
  }

  // NOTIFICATION OPERATIONS
  Future<AppNotification> saveNotification(AppNotification notification) async {
    final db = await database;
    final id = await db.insert('notifications', {
      'userId': notification.userId,
      'title': notification.title,
      'body': notification.body,
      'type': notification.type,
      'isRead': notification.isRead ? 1 : 0,
      'createdAt': notification.createdAt.toIso8601String(),
      'relatedUserId': notification.relatedUserId,
    });
    return notification.copyWith(id: id);
  }

  Future<List<AppNotification>> getUserNotifications(int userId) async {
    final db = await database;
    final result = await db.query(
      'notifications',
      where: 'userId = ?',
      whereArgs: [userId],
      orderBy: 'createdAt DESC',
    );
    return result.map((json) => AppNotification.fromJson(json)).toList();
  }

  Future<void> markNotificationAsRead(int notificationId) async {
    final db = await database;
    await db.update(
      'notifications',
      {'isRead': 1},
      where: 'id = ?',
      whereArgs: [notificationId],
    );
  }
}

import 'dart:convert';
import 'package:web_socket_channel/web_socket_channel.dart';
import 'storage_service.dart';
import '../screens/debug_screen.dart';

class WebSocketService {
  static WebSocketChannel? _channel;
  static bool _isConnected = false;
  
  // Callbacks for different events
  static Function(Map<String, dynamic>)? onMessageReceived;
  static Function(int, String)? onUserOnline;
  static Function(int)? onUserOffline;
  
  // ✅ Connect to WebSocket
  static Future<void> connect(int userId, String username) async {
    try {
      final baseUrl = StorageService.getServerUrl();
      addDebugMessage('═══════════════════════════════════════');
      addDebugMessage('🔌 WebSocket CONNECTING');
      addDebugMessage('Base URL: $baseUrl');
      
      final wsUrl = baseUrl
          .replaceFirst('http://', 'ws://')
          .replaceFirst('https://', 'wss://');
      
      final fullUrl = '$wsUrl/ws-chat';
      addDebugMessage('Full WebSocket URL: $fullUrl');
      
      _channel = WebSocketChannel.connect(
        Uri.parse(fullUrl),
      );
      
      _isConnected = true;
      addDebugMessage('✅ WebSocket Connected!');
      
      // Send login message
      _sendLoginMessage(userId, username);
      
      // Listen for messages
      _channel?.stream.listen(
        (message) {
          addDebugMessage('📨 WebSocket RAW Message: $message');
          try {
            final decoded = jsonDecode(message);
            addDebugMessage('📨 WebSocket DECODED: $decoded');
            _handleMessage(decoded);
          } catch (e) {
            addDebugMessage('❌ Error decoding message: $e');
          }
        },
        onError: (error) {
          addDebugMessage('❌ WebSocket Error: $error');
          _isConnected = false;
        },
        onDone: () {
          addDebugMessage('🔌 WebSocket Closed');
          _isConnected = false;
        },
      );
      
    } catch (e) {
      addDebugMessage('❌ Error connecting to WebSocket: $e');
      _isConnected = false;
    }
  }
  
  // ✅ Send login message
  static void _sendLoginMessage(int userId, String username) {
    if (!_isConnected) {
      addDebugMessage('⚠️ WebSocket not connected, cannot send login');
      return;
    }
    
    final message = {
      'type': 'login',
      'senderId': userId,
      'senderName': username,
      'timestamp': DateTime.now().millisecondsSinceEpoch,
    };
    
    addDebugMessage('📤 Sending LOGIN message: $message');
    _channel?.sink.add(jsonEncode(message));
  }
  
  // ✅ Send chat message
  static void sendMessage(int senderId, int receiverId, String content, String senderName) {
    if (!_isConnected) {
      addDebugMessage('⚠️ WebSocket not connected, cannot send message');
      return;
    }
    
    final message = {
      'type': 'message',
      'senderId': senderId,
      'receiverId': receiverId,
      'content': content,
      'senderName': senderName,
      'timestamp': DateTime.now().millisecondsSinceEpoch,
    };
    
    addDebugMessage('📤 Sending MESSAGE via WebSocket: $message');
    _channel?.sink.add(jsonEncode(message));
  }
  
  // ✅ Send online status
  static void setOnline(int userId, String username) {
    if (!_isConnected) {
      addDebugMessage('⚠️ WebSocket not connected, cannot send online');
      return;
    }
    
    final message = {
      'type': 'online',
      'senderId': userId,
      'senderName': username,
      'timestamp': DateTime.now().millisecondsSinceEpoch,
    };
    
    addDebugMessage('📤 Sending ONLINE status');
    _channel?.sink.add(jsonEncode(message));
  }
  
  // ✅ Send offline status
  static void setOffline(int userId) {
    if (!_isConnected) {
      addDebugMessage('⚠️ WebSocket not connected, cannot send offline');
      return;
    }
    
    final message = {
      'type': 'offline',
      'senderId': userId,
      'timestamp': DateTime.now().millisecondsSinceEpoch,
    };
    
    addDebugMessage('📤 Sending OFFLINE status');
    _channel?.sink.add(jsonEncode(message));
  }
  
  // ✅ Handle incoming messages
  static void _handleMessage(Map<String, dynamic> message) {
    addDebugMessage('🎯 Handling WebSocket message: $message');
    
    final type = message['type'];
    
    switch (type) {
      case 'message':
        addDebugMessage('💬 Message callback triggered');
        onMessageReceived?.call(message);
        break;
        
      case 'user_online':
        final userId = message['senderId'];
        final username = message['senderName'];
        addDebugMessage('✅ User online callback: $username (ID: $userId)');
        onUserOnline?.call(userId, username);
        break;
        
      case 'user_offline':
        final userId = message['senderId'];
        addDebugMessage('❌ User offline callback: $userId');
        onUserOffline?.call(userId);
        break;
        
      default:
        addDebugMessage('❓ Unknown message type: $type');
    }
  }
  
  // ✅ Check if connected
  static bool isConnected() {
    return _isConnected;
  }
  
  // ✅ Disconnect
  static void disconnect() {
    addDebugMessage('🔌 Disconnecting WebSocket');
    _channel?.sink.close();
    _isConnected = false;
  }
}
